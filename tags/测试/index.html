- (void)longPress:(UILongPressGestureRecognizer *)press {
    if (press.state == UIGestureRecognizerStateBegan) {
        int index = (int)press.view.tag - KIMGTAG;
        if (index < 0 || index >= _smallImgs.count) {
            return;
        }
        [_smallImgs removeObjectAtIndex:index];
        [self updateAll];
    }
}
